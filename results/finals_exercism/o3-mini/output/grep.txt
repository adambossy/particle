def matches(line, pattern, flags):
    if '-i' in flags:  # case-insensitive
        line = line.lower()
        pattern = pattern.lower()

    if '-x' in flags:  # match entire lines
        if len(pattern) != len(line.rstrip()):
            return '-v' in flags

    if '-v' in flags:  # invert matching
        return pattern not in line

    return pattern in line


def format_files(matched_lines):
    result = ''

    for file_name, _, _ in matched_lines:
        if file_name not in result:
            result += file_name + '\n'

    return result


def format_lines(matched_lines, flags, files):
    result = []

    for file_name, line_number, line in matched_lines:
        line_result = ''

        if len(files) > 1:
            line_result += file_name + ':'

        if '-n' in flags:
            line_result += str(line_number) + ':'

        line_result += line

        result.append(line_result)

    return ''.join(result)


def grep(pattern, flags, files):
    matched_lines = []

    for file_name in files:
        with open(file_name, encoding='utf-8') as f:
            for line_number, line in enumerate(f.readlines(), start=1):
                if matches(line, pattern, flags):
                    matched_lines.append((file_name, line_number, line))

    if '-l' in flags:
        return format_files(matched_lines)

    return format_lines(matched_lines, flags, files)

package meta

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

// match represents a matched line in a file.
type match struct {
	fileName   string
	lineNumber int
	line       string
}

// containsFlag checks if a specific flag exists in the flags slice.
func containsFlag(flags []string, flag string) bool {
	for _, f := range flags {
		if f == flag {
			return true
		}
	}
	return false
}

// matches determines if a line matches the given pattern with the provided flags.
// It implements case-insensitive matching (-i), full-line matching (-x), and inverted matching (-v).
func matches(line string, pattern string, flags []string) bool {
	// Handle case-insensitive flag
	if containsFlag(flags, "-i") {
		line = strings.ToLower(line)
		pattern = strings.ToLower(pattern)
	}

	// Handle full line matching (-x)
	if containsFlag(flags, "-x") {
		// In Go, line is already trimmed of newline by scanner so we mimic rstrip
		if len(pattern) != len(strings.TrimRight(line, "\n")) {
			return containsFlag(flags, "-v")
		}
	}

	// Handle invert match (-v)
	if containsFlag(flags, "-v") {
		return !strings.Contains(line, pattern)
	}

	return strings.Contains(line, pattern)
}

// formatFiles returns a slice of unique file names from the matched lines.
// Each file name appears only once in the order of first occurrence.
func formatFiles(matchedLines []match) []string {
	var result []string
	seen := make(map[string]bool)
	for _, m := range matchedLines {
		if !seen[m.fileName] {
			result = append(result, m.fileName)
			seen[m.fileName] = true
		}
	}
	return result
}

// formatLines returns a slice of formatted matching lines based on flags and input files.
// If multiple files are searched, the result includes the file name as a prefix.
// If the -n flag is present, the line number is prefixed as well.
func formatLines(matchedLines []match, flags []string, files []string) []string {
	var result []string
	multipleFiles := len(files) > 1
	for _, m := range matchedLines {
		lineResult := ""
		if multipleFiles {
			lineResult += m.fileName + ":"
		}
		if containsFlag(flags, "-n") {
			lineResult += fmt.Sprintf("%d:", m.lineNumber)
		}
		lineResult += m.line
		result = append(result, lineResult)
	}
	return result
}

// Search searches for the given pattern in the provided files applying the specified flags.
// It returns a slice of strings that contain either the file names (if -l flag is used) or
// the formatted matching lines.
func Search(pattern string, flags, files []string) []string {
	var matchedLines []match

	for _, fileName := range files {
		file, err := os.Open(fileName)
		if err != nil {
			// For simplicity, if a file cannot be opened, we panic.
			panic(fmt.Sprintf("failed to open file %s: %v", fileName, err))
		}
		defer file.Close()

		scanner := bufio.NewScanner(file)
		lineNumber := 1
		for scanner.Scan() {
			line := scanner.Text()
			if matches(line, pattern, flags) {
				matchedLines = append(matchedLines, match{
					fileName:   fileName,
					lineNumber: lineNumber,
					line:       line,
				})
			}
			lineNumber++
		}
		if err := scanner.Err(); err != nil {
			panic(fmt.Sprintf("error reading file %s: %v", fileName, err))
		}
	}

	if containsFlag(flags, "-l") {
		return formatFiles(matchedLines)
	}

	return formatLines(matchedLines, flags, files)
}
