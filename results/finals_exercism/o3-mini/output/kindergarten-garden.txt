class Garden:

    STUDENTS = [
        'Alice',
        'Bob',
        'Charlie',
        'David',
        'Eve',
        'Fred',
        'Ginny',
        'Harriet',
        'Ileana',
        'Joseph',
        'Kincaid',
        'Larry',
    ]
    PLANTS = {'C': 'Clover', 'G': 'Grass', 'R': 'Radishes', 'V': 'Violets'}

    def __init__(self, diagram, students=None):
        students = sorted(students or self.STUDENTS)
        front, back = diagram.splitlines()
        self.cups = {}
        for idx, student in enumerate(students[: len(front)]):
            start = idx * 2
            stop = start + 2
            self.cups.setdefault(student, [])
            self.cups[student].extend(
                self.PLANTS[plant] for plant in front[start:stop]
            )
            self.cups[student].extend(
                self.PLANTS[plant] for plant in back[start:stop]
            )

    def plants(self, student):
        return self.cups.get(student, [])

import (
	"errors"

// plantMap maps a plant's initial letter to its full name.
var plantMap = map[rune]string{
	'C': "Clover",
	'G': "Grass",
	'R': "Radishes",
	'V': "Violets",
}

// defaultStudents is the default list of students.
var defaultStudents = []string{
	"Alice",
	"Bob",
	"Charlie",
	"David",
	"Eve",
	"Fred",
	"Ginny",
	"Harriet",
	"Ileana",
	"Joseph",
	"Kincaid",
	"Larry",
}

// Garden represents a garden with cups assigned to each student.
// Each cup contains a plant, and each student receives 4 cups (2 per row) as per diagram.
// The diagram argument starts each row with a '\n'. This allows Go's raw string literals to present diagrams
// in source code nicely as two rows flush left, for example,
//
//	diagram := `
//	VVCCGG
//	VVCCGG`

// NewGarden creates a new Garden from the given diagram and list of children.
// If the children slice is nil or empty, the default student list is used.
// Returns an error if the diagram does not consist of exactly 2 rows or if the rows do not
// have an even number of cups, or if there are not enough student names provided.
func NewGarden(diagram string, children []string) (*Garden, error) {
	// Use default students if none provided
	if len(children) == 0 {
		// Make a copy of defaultStudents to avoid mutating the global slice
		children = append([]string(nil), defaultStudents...)
	} else {
		// Copy provided slice to avoid side effects
		children = append([]string(nil), children...)
	}

	sort.Strings(children)

	// Trim and split the diagram into two lines
	lines := strings.Split(strings.TrimSpace(diagram), "\n")
	if len(lines) != 2 {
		return nil, errors.New("diagram must consist of exactly 2 rows")
	}
	front := lines[0]
	back := lines[1]

	// Ensure each row has an even number of cups
	if len(front)%2 != 0 || len(back)%2 != 0 {
		return nil, errors.New("each row in diagram must have an even number of cups")
	}

	// Determine the number of students from the diagram
	numStudents := len(front) / 2
	if numStudents > len(children) {
		return nil, errors.New("not enough student names provided for the number of cups")
	}

	cups := make(map[string][]string)
	for i := 0; i < numStudents; i++ {
		student := children[i]
		cups[student] = make([]string, 0, 4)
		start := i * 2

		// Process the front row
		for j := start; j < start+2; j++ {
			if plant, ok := plantMap[rune(front[j])]; ok {
				cups[student] = append(cups[student], plant)
			}
		}
		
		// Process the back row
		for j := start; j < start+2; j++ {
			if plant, ok := plantMap[rune(back[j])]; ok {
				cups[student] = append(cups[student], plant)
			}
		}
	}

	return &Garden{cups: cups}, nil
}

// Garden represents a garden with cups assigned to each student.
type Garden struct {
	cups map[string][]string
}

// Plants returns the slice of plant names for the given student
// and a boolean indicating whether the student was found in the garden.
func (g *Garden) Plants(child string) ([]string, bool) {
	plants, ok := g.cups[child]
	return plants, ok
}

// NewGarden creates a new Garden from the given diagram and children.
// The diagram must start with a newline and consist of exactly 2 rows.
// Each row must have the same even number of cups.
// The children slice is copied and sorted alphabetically.
// If children is nil or empty, the default student list is used.
// Returns an error if the diagram format is invalid, if the rows have odd lengths or mismatched lengths,
// if there are duplicate names in the children, if there are not enough student names provided,
// or if an invalid cup code is encountered in the diagram.
func NewGarden(diagram string, children []string) (*Garden, error) {
	// The diagram must start with a newline
	if len(diagram) == 0 || diagram[0] != '\n' {
		return nil, errors.New("diagram must start with a newline")
	}

	// Remove the first newline and split into rows
	trimmed := strings.TrimPrefix(diagram, "\n")
	lines := strings.Split(trimmed, "\n")
	if len(lines) != 2 {
		return nil, errors.New("diagram must consist of exactly 2 rows")
	}

	front, back := lines[0], lines[1]

	// Check if rows have the same length
	if len(front) != len(back) {
		return nil, errors.New("diagram rows do not have the same length")
	}

	// Check that each row has an even number of cups
	if len(front)%2 != 0 {
		return nil, errors.New("each row in diagram must have an even number of cups")
	}

	// Use default students if none provided
	if len(children) == 0 {
		children = append([]string(nil), defaultStudents...)
	} else {
		children = append([]string(nil), children...)
	}

	// Sort the children to get alphabetical order
	sortedChildren := append([]string(nil), children...)
	sort.Strings(sortedChildren)

	// Check for duplicates
	for i := 1; i < len(sortedChildren); i++ {
		if sortedChildren[i] == sortedChildren[i-1] {
			return nil, errors.New("duplicate student name found")
		}
	}

	numStudents := len(front) / 2
	if numStudents > len(sortedChildren) {
		return nil, errors.New("not enough student names provided for the number of cups")

	for i := 0; i < numStudents; i++ {
		student := sortedChildren[i]
		cups[student] = make([]string, 0, 4)
		start := i * 2

		// Process cups from the front row
		for j := start; j < start+2; j++ {
			code := rune(front[j])
			plant, ok := plantMap[code]
			if !ok {
				return nil, errors.New("invalid cup code")
			}
			cups[student] = append(cups[student], plant);
		}

		// Process cups from the back row
		for j := start; j < start+2; j++ {
			code := rune(back[j])
			plant, ok := plantMap[code]
			if !ok {
				return nil, errors.New("invalid cup code")
			}
			cups[student] = append(cups[student], plant);
		}
	}

	return &Garden{cups: cups}, nil
}
