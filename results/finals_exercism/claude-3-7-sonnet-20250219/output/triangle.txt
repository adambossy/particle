def valid(sides):
    return sum(sorted(sides)[:2]) >= sorted(sides)[2] and all(
        side > 0 for side in sides
    )


def equilateral(sides):
    return valid(sides) and all(sides[0] == side for side in sides)


def isosceles(sides):
    return valid(sides) and any(
        side_1 == side_2 for side_1, side_2 in zip(sorted(sides), sorted(sides)[1:])
    )


def scalene(sides):
    return valid(sides) and not isosceles(sides)

// exercises/practice/triangle/.meta/example.go
package triangle

import "math"

// Kind represents the type of triangle.
type Kind int

const (
	NaT Kind = iota // not a triangle
	Equ             // equilateral
	Iso             // isosceles
	Sca             // scalene
)

// valid checks if the sides can form a valid triangle.
func valid(a, b, c float64) bool {
	// Check for positive values and NaN values
	if a <= 0 || b <= 0 || c <= 0 ||
		math.IsNaN(a) || math.IsNaN(b) || math.IsNaN(c) ||
		math.IsInf(a, 0) || math.IsInf(b, 0) || math.IsInf(c, 0) {
		return false
	}

	// Triangle inequality theorem
	return a+b >= c && b+c >= a && a+c >= b
}

// KindFromSides determines the kind of triangle from the lengths of its sides.
func KindFromSides(a, b, c float64) Kind {
	if !valid(a, b, c) {
		return NaT
	}
	
	// Equilateral: all sides equal
	if a == b && b == c {
		return Equ
	}
	
	// Isosceles: at least two sides equal
	if a == b || b == c || a == c {
		return Iso
	}
	
	// Scalene: no sides equal
	return Sca
}
